#include <bits/stdc++.h>
using namespace std;


 int main(){

int n; cin >>n;
double a[n],tong=0,min = INT_MAX,max= INT_MIN;
for(int i =0 ; i< n; i++){cin >>a[i];
tong += a[i]/n;
if(a[i] < min){min=a[i];}
if(a[i] > max){max =a[i];}
}
 cout << "Mean: "<< tong <<endl
 << "Max: "<< max << endl
<< "Min: " << min ;
 }
